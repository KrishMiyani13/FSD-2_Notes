    <script>
        const numerals = ["I", "II", "III", "IV", "V", "VI", "VII", "VIII", "IX", "X", "XI", "XII"];
        const radius = 120;

        function placeNumerals(groupId) {
            const group = document.getElementById(groupId);
            numerals.forEach((num, i) => {
                const angle = (i * 30 - 60) * Math.PI / 180;
                const x = 150 + radius * Math.cos(angle);
                const y = 150 + radius * Math.sin(angle);
                const text = document.createElementNS("http://www.w3.org/2000/svg", "text");
                text.setAttribute("x", x);
                text.setAttribute("y", y);
                text.textContent = num;
                group.appendChild(text);
            });
        }

        placeNumerals("numeralsIST");
        placeNumerals("numeralsLondon");

        function updateClock() {
            const now = new Date();

            // Indian Standard Time using Intl API
            const istDate = new Date(now.toLocaleString("en-US", { timeZone: "Asia/Kolkata" }));

            // London Time using Intl API
            const londonDate = new Date(now.toLocaleString("en-US", { timeZone: "Europe/London" }));

            function setHands(date, prefix) {
                const ms = date.getMilliseconds();
                const seconds = date.getSeconds();
                const minutes = date.getMinutes();
                let hours = date.getHours();

                const secondAngle = seconds * 6 + ms * 0.006;
                const minuteAngle = minutes * 6 + seconds * 0.1;
                const hourAngle = (hours % 12) * 30 + minutes * 0.5;

                document.getElementById("second" + prefix).style.transform = `rotate(${secondAngle}deg)`;
                document.getElementById("minute" + prefix).style.transform = `rotate(${minuteAngle}deg)`;
                document.getElementById("hour" + prefix).style.transform = `rotate(${hourAngle}deg)`;

                // Convert to 12-hour format with AM/PM
                const ampm = hours >= 12 ? "PM" : "AM";
                hours = hours % 12;
                hours = hours ? hours : 12; // 0 becomes 12

                const digital =
                    String(hours).padStart(2, "0") + ":" +
                    String(minutes).padStart(2, "0") + ":" +
                    String(seconds).padStart(2, "0") + " " + ampm;
                document.getElementById("digital" + prefix).textContent = digital;
            }

            setHands(istDate, "IST");
            setHands(londonDate, "London");
        }

        setInterval(updateClock, 1000); // update every second
        updateClock();
    </script>
